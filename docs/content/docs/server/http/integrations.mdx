## Hono

```ts
import { createFetch } from '@xsmcp/server-http'
import { createServer } from '@xsmcp/server-shared'
import { Hono } from 'hono' // [!code highlight]

import * as tools from '...'

const server = createServer({ ...options })

for (const tool of tools) {
  server.addTool(tool)
}

const fetch = createFetch(server)

const app = new Hono() // [!code highlight]
  .post('/mcp', ({ req }) => fetch(req.raw)) // [!code highlight]
```
