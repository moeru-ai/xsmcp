
```ts twoslash
import { createHttpServer } from '@xsmcp/server-http'
import { createServer as createMcpServer } from '@xsmcp/server-shared'
import { createServer } from 'node:http'

const s = createMcpServer({
  name: 'example-server',
  version: '1.0.0'
})

const server = createHttpServer(s)

export const GET = server.fetch // [!code highlight]
export const POST = server.fetch // [!code highlight]
```
